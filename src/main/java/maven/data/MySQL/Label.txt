create database Label;

use Label;

create table `iLabel`(
`UserId` varchar(20),
`TaskId` varchar(100),
`iNumber` integer,
`Content` varchar(100),
primary key(UserId, TaskId, iNumber)
);

create table `aLabel`(
`UserId` varchar(20),
`TaskId` varchar(100),
`iNumber` integer,
`aNumber` integer,
`Content` varchar(1000),
primary key(userId, TaskId, iNumber, aNumber)
);

create table `fLabel`(
`UserId` varchar(20),
`TaskId` varchar(100),
`iNumber` integer,
`fNumber` integer,
`Content` varchar(1000),
primary key(UserId, TaskId, iNumber, fNumber)
);
